//Compute the product of an array except self.
#include <stdio.h>

void product(int arr[], int size, int result[]) {
    int left[size];
    int right[size];

    left[0] = 1;
    right[size - 1] = 1;

    
    for (int i = 1; i < size; i++) 
    {
        left[i] = left[i - 1] * arr[i - 1];
    }

    // Calculate the product of elements to the right of the current index
    for (int i = size - 2; i >= 0; i--) 
    {
        right[i] = right[i + 1] * arr[i + 1];
    }

    // Calculate the final result by multiplying left and right products
    for (int i = 0; i < size; i++)
    {
        result[i] = left[i] * right[i];
    }
    printf("The product array except self: ");
    for (int i = 0; i < size; i++) 
    {
        printf("%d ", result[i]);
    }

}

int main() {
    int arr[] = {1, 2, 3, 4};
    int size = sizeof(arr) / sizeof(arr[0]);
    int result[size];
    product(arr, size, result);
}
